üöÄ Using Both Widgets Together

Combine age gate + chatbot for the ultimate customer experience:

<!-- Age gate appears first, chatbot loads after verification -->
<link rel="stylesheet" href="https://bakedbot.ai/embed/age-gate.css">
<link rel="stylesheet" href="https://bakedbot.ai/embed/chatbot.css">

<script>
    // Age gate configuration
    window.BakedBotAgeGateConfig = {
        brandId: 'ecstatic-edibles',
        state: 'IL',
        source: 'ecstaticedibles.com'
    };

    // Chatbot configuration
    window.BakedBotConfig = {
        brandId: 'ecstatic-edibles',
        entityName: 'Ecstatic Edibles',
        primaryColor: '#8b5cf6',
        botName: 'Smokey'
    };
</script>

<script src="https://bakedbot.ai/embed/age-gate.js"></script>
<script src="https://bakedbot.ai/embed/chatbot.js"></script>
